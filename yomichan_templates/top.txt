{{~! REGEX ~}}
{{~! matches most JMdict dictionaries and 新和英 ~}}
{{~#set "bilingual-dict-regex"~}} ^(([Jj][Mm][Dd]ict)(?! Surface Forms)(.*)|新和英.*|日本語文法辞典.*|ADD_BILINGUAL_DICTIONARIES_HERE)(\[object Object\])?$ {{~/set~}}
{{~#set "utility-dict-regex"~}} ^(NHK.*|シン・漢字遣い参考|JMDict Surface Forms|ADD_UTILITY_DICTIONARIES_HERE)(\[object Object\])?$ {{~/set~}}
{{~#set "ignored-dict-regex"~}} ^(ADD_IGNORED_DICTIONARIES_HERE)(\[object Object\])?$ {{~/set~}}

{{~! OPTIONS ~}}

{{~! valid values: "bilingual", "monolingual" ~}}
{{~#set "opt-first-definition-type" "bilingual"}}{{/set~}}

{{~! valid values: 0, 1, 2, 3
disabled:
- selection text is entirely ignored
{{~#set "opt-selection-text-enabled" false}}{{/set~}}

enabled:
    - Attempts to search for dictionary within the selected text
    - If the selection is equal to the dictionary name -> uses that dictionary as the primary definition
    - If a dictionary is not contained -> simply uses the selected text in place of the primary definition
        - The dictionary that the selected text will not be detected, so the same dictionary
          should appear in one of the SecondaryDefinition or ExtraDefinition fields.
~}}
{{~#set "opt-selection-text-enabled" false}}{{/set~}}
